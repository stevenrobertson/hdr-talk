class YUVSpace(object):
    def __init__(self, Kr, Kg, Kb, Knb, Knr):
        self.Kr, self.Kg, self.Kb, self.Knb, self.Knr = Kr, Kg, Kb, Knb, Knr
        
        self.rgb2yuv_matrix = np.mat([[self.Kr, self.Kg, self.Kb],
                                      np.array([-self.Kr, -self.Kg, 1 - self.Kb]) / self.Knb,
                                      np.array([1 - self.Kr, -self.Kg, -self.Kb]) / self.Knr])
        self.yuv2rgb_matrix = self.rgb2yuv_matrix.I

    def rgb2yuv(self, r, g, b):
        y = r * self.Kr + g * self.Kg + b * self.Kb
        u = (b - y) / self.Knb
        v = (r - y) / self.Knr
        return (y, u, v)
        
    def yuv2rgb(self, y, u, v):
        b = self.Knb * u + y
        r = self.Knr * v + y
        g = (y - self.Kr * r - self.Kb * b) / self.Kg
        return r, g, b

class SMPTE2084TransferFunc(object):
    name = 'smpte-st-2084'
    def __init__(self, luminance_cap=1000):
        self.luma = luminance_cap
        self.factor = luminance_cap / 10000.
        
    def oetf(self, v):
        ve = (v * self.factor) ** 0.1593017578125
        return ((0.8359375 + 18.8515625 * ve) / (1 + 18.6875 * ve)) ** 78.84375

    def eotf(self, v):
        ve = v ** (1 / 78.84375)
        return (max(0, ve - 0.8359375) / (18.8515625 - 18.6875 * ve)) ** (1 / 0.1593017578125) / self.factor

class GammaTransferFunc(object):
    luma = 400
    name = 'bt709'
    def __init__(self, gamma=2.4):
        self.gamma = gamma

    def oetf(self, v):
        return v ** (1 / self.gamma)
    
    def eotf(self, v):
        return v ** self.gamma
    
class ARIBStdB67TransferFunc(object):
    name = 'arib-std-b67'
    def __init__(self, system_gamma=1.25):
        # TODO: auto-derive params from Lmax
        self.system_gamma = system_gamma
    
    luma = 1000  # dud value
    a = 0.17883277
    b = 0.28466892
    c = 0.55991073
    Lmax = 12
    
    def oetf(self, v):
        # Note that normally we wouldn't apply system gamma to the input, but 
        # since we are after all doing gamma tests...
        E = v ** (1 / self.system_gamma) * self.Lmax
        if E <= 1:
            E_ = 0.5 * np.sqrt(E)
        else:
            E_ = self.a * np.log(E - self.b) + self.c
        return E_
    
    def eotf(self, E_):
        if E_ <= 0.5:
            v_ = (E_ * 2) ** 2
        else:
            v_ = np.exp((E_ - self.c) / self.a) + self.b
        return (v_ / self.Lmax) ** self.system_gamma
    
class CachingTransferFunc(object):
    def __init__(self, base):
        self.base = base
        self.cache = {}
    def oetf(self, v):
        if v not in self.cache:
            self.cache[v] = self.base.oetf(v)
        return self.cache[v]
    def eotf(self, v):
        return self.base.eotf(v)

def display_inline(s):
    # My display apparently uses a gamma of 2.2. This won't affect the output at
    # all, just the inline display.
    scaled = np.uint8((np.flipud(s) ** (1/2.2)) * 255)
    img = PIL.Image.frombuffer('RGB', (scaled.shape[1], scaled.shape[0]), scaled.data)
    sio = cStringIO.StringIO()
    img.save(sio, 'png')
    sio.seek(0)
    return display.Image(sio.read())

class Slate(object):
    def __init__(self, w=3840, h=2160):
        self.w, self.h = w, h
        self.data = np.zeros((h, w, 3))
        
        self.img = PIL.Image.new('RGB', (w,h))
        self.draw = PIL.ImageDraw.Draw(self.img)
        try:
            self.font = PIL.ImageFont.truetype(
                '/usr/share/fonts/truetype/droid/DroidSansMono.ttf', 16)
        except:
            self.font = None
        
    def merge_overlay(self):
        overlay_data = np.frombuffer(self.img.tostring(), 'u1').reshape((self.h, self.w, 3))
        return self.data + overlay_data / 512.
        
default_gamut_tests = {'709': test709, 'P3': testP3}
        
def gamma_test(s, source=(0,0,0), target=(1,1,1), steps=14, checker_to_step=5.0, x=32, y=32,
               width=256, gamut_tests=default_gamut_tests, checker_2x=False, shift_lab=False):
    """
    Draw a gamma ramp with some properties into a linear light array.
    
    s: Slate to draw into.
    target: Color endpoint (highest value that will be output, assuming width 2.0).
    steps:  Number of steps to draw. Note that, because we'll go +/- two steps to draw
            the checker pattern, this should probably be of the form 2**x-4.
    checker_to_step: Ratio of the size of the luma swing relative to a step size.
            Values greater than 2.0 will clip until I fix that.
    x:      Start pixel. Labels will be written below this, so ensure clearance.
    y:      Start pixel.
    gamut_tests: Dict mapping from short gamut names to functions
            that test whether a 3-value is in-gamut.
    shift_lab: If True, instead of adjusting the linear-light luminance, the alternate
            colors will be formed by shifting the a*b* coordinates by that amount.
    """
    
    stepsize = 1 / (steps + 4.)
    source = np.array(source)
    target = np.array(target)
    
    w2 = width / 2
    for step in range(steps):
        val = (step + 1) * stepsize
        x_start = x + step * width
        x_end = x + (step + 1) * width
        mid_color = source * (1 - val) + target * val
        s.data[y:y+w2,x_start:x_end,:] = mid_color
        
        
        
        if shift_lab:
            csp = colour.REC_2020_COLOURSPACE
            val_xyz = np.array(csp.RGB_to_XYZ_matrix.dot(mid_color))
            illum = colour.ILLUMINANTS['CIE_2_1931']['D65']
            
            val_lab = colour.XYZ_to_Lab(val_xyz, illum)
            lab_shift = stepsize * checker_to_step * 50
            hi_lab = [val_lab[0], val_lab[1] + lab_shift, val_lab[2] + lab_shift]
            lo_lab = [val_lab[0], val_lab[1] - lab_shift, val_lab[2] - lab_shift]
            hi_color = np.array(csp.XYZ_to_RGB_matrix.dot(colour.Lab_to_XYZ(hi_lab, illum)))
            lo_color = np.array(csp.XYZ_to_RGB_matrix.dot(colour.Lab_to_XYZ(lo_lab, illum)))
            #hi_color = Rec601YUV.yuv2rgb(*Rec709YUV.rgb2yuv(*hi_color))
            #lo_color = Rec601YUV.yuv2rgb(*Rec709YUV.rgb2yuv(*lo_color))
        else:
            lo_val = val * (1 - stepsize * checker_to_step)
            hi_val = val * (1 + stepsize * checker_to_step)
            hi_color = source * (1 - hi_val) + target * hi_val
            lo_color = source * (1 - lo_val) + target * lo_val
        
        if checker_2x:
            for j in range(width/2):
                for i in range(width):
                    s.data[y+w2+j,x_start+i,:] = hi_color if (j & 2) == (i & 2) else lo_color
        else:
            # TODO: do error diffusion by roundtripping through EOTF and clipping
            s.data[y+w2:y+width:2,x_start:x_end:2,:] = hi_color
            s.data[y+w2+1:y+width:2,x_start+1:x_end:2,:] = hi_color
            s.data[y+w2+1:y+width:2,x_start:x_end:2,:] = lo_color
            s.data[y+w2:y+width:2,x_start+1:x_end:2,:] = lo_color
            
        if gamut_tests:
            gamut_passed = [k for k, v in gamut_tests.items() if v(hi_color)]
            if gamut_passed and s.font:
                s.draw.text((x_start,y+width), ','.join(sorted(gamut_passed)), 
                            font=s.font)
            
        
def luma_ramp(w, h, boxsize):
    s = Slate(w, h)
    gamma_test(s, y=64, width=boxsize)
    gamma_test(s, y=64 + (boxsize + 32) * 1, target=(1,0,0), width=boxsize)
    gamma_test(s, y=64 + (boxsize + 32) * 2, target=(1,1,0), width=boxsize)
    gamma_test(s, y=64 + (boxsize + 32) * 3, target=(0,1,0), width=boxsize)
    gamma_test(s, y=64 + (boxsize + 32) * 4, target=(0,1,1), width=boxsize)
    gamma_test(s, y=64 + (boxsize + 32) * 5, target=(0,0,1), width=boxsize)
    gamma_test(s, y=64 + (boxsize + 32) * 6, target=(1,0,1), width=boxsize)
    return s

def lut_3d_cubic(lut, y, u, v, shift=5):
    yl, ul, vl = y >> shift, u >> shift, v >> shift
    width = 2. ** shift
    mask = (2 ** shift) - 1
    yhs, uhs, vhs = (y & mask) / width, (u & mask) / width, (v & mask) / width
    yls, uls, vls = 1 - yhs, 1 - uhs, 1 - vhs
    
    out = (lut[yl  ,ul  ,vl  ] * yls * uls * vls +
           lut[yl+1,ul  ,vl  ] * yhs * uls * vls +
           lut[yl  ,ul+1,vl  ] * yls * uhs * vls +
           lut[yl+1,ul+1,vl  ] * yhs * uhs * vls +
           lut[yl  ,ul  ,vl+1] * yls * uls * vhs +
           lut[yl+1,ul  ,vl+1] * yhs * uls * vhs +
           lut[yl  ,ul+1,vl+1] * yls * uhs * vhs +
           lut[yl+1,ul+1,vl+1] * yhs * uhs * vhs)
    return np.round(out)

def diff_lut(testdata, testfunc, lut, shift=5):
    count = 0
    sqerror = 0
    for y, u, v in testdata:
        funcval = testfunc(y,u,v)
        lutval = lut_3d_cubic(lut, y, u, v, shift)
        err = funcval - lutval
        sqerror += err * err
        count += 1
    return sqerror / count

class FnEvaluator(ast.NodeVisitor):
    def __init__(self, functions):
        self.assigns = {}
        self.functions = functions
    def visit(self, val):
        if isinstance(val, basestring):
            return val
        else:
            return super(FnEvaluator, self).visit(val)
    def visit_Assign(self, assign):
        self.assigns[assign.targets[0].id] = assign.value
        assert(assign.value is not None)
    def visit_Return(self, ret):
        return self.visit(ret.value)
    def visit_Call(self, call):
        assert isinstance(call.func, ast.Name)
        if call.func.id in self.functions:
            return FnEvaluator(self.functions).eval_fn(call.func.id, map(self.visit, call.args))
        else:
            return call.func.id + '(' + ','.join(map(self.visit, call.args)) + ')'
    def visit_BinOp(self, binop):
        if isinstance(binop.op, ast.Pow):
            return 'pow(%s,%s)' % (self.visit(binop.left), self.visit(binop.right))
        op = ({
            ast.Add: '+',
            ast.Sub: '-',
            ast.Mult: '*',
            ast.Div: '/'
        })[type(binop.op)]
        return '(' + self.visit(binop.left) + op + self.visit(binop.right) + ')'
    def visit_Num(self, num):
        return str(num.n)
    def visit_Name(self, name):
        return self.visit(self.assigns[name.id])
    def visit_IfExp(self, ifexp):
        return 'if(%s,%s,%s)' % (self.visit(ifexp.test), self.visit(ifexp.body), self.visit(ifexp.orelse))
    def visit_Compare(self, cmp):
        assert len(cmp.ops) == 1
        op = ({
            ast.Eq: 'eq',
            ast.NotEq: 'neq',
            ast.Lt: 'lt',
            ast.LtE: 'lte',
            ast.Gt: 'gt',
            ast.GtE: 'gte',
        })[type(cmp.ops[0])]
        return '%s(%s,%s)' % (op, self.visit(cmp.left), self.visit(cmp.comparators[0]))
    
    def generic_visit(self, node):
        raise NotImplementedError(str(node))
        
    def eval_fn(self, name, argvals):
        fn = self.functions[name]
        self.assigns = {}
        for arg, val in zip(fn.args.args, argvals):
            assert isinstance(arg, ast.Name)
            self.assigns[arg.id] = val
        for stmt in fn.body:
            out = self.visit(stmt)
        return out
